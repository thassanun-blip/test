        let fullData = new Map();

        // --- DATA GENERATION ---
        function generateMockData() {
            dimensions.forEach(dim => {
                dim.indicators.forEach(ind => {
                    const indicatorData = [];
                    provinces.forEach(province => {
                        const [min, max] = ind.range;
                        const rawValue = Math.random() * (max - min) + min;
                        const value = ind.unit === "%" || ind.id === "2.5" ? parseFloat(rawValue.toFixed(1)) : Math.round(rawValue);
                        
                        // Calculate score from 0-100
                        let score;
                        if (ind.higherIsBetter) {
                            score = Math.round(((value - min) / (max - min)) * 80) + 10; // Score between 10-90
                        } else {
                            score = Math.round(((max - value) / (max - min)) * 80) + 10; // Score between 10-90
                        }

                        indicatorData.push({ province, value, score });
                    });
                    fullData.set(ind.id, indicatorData);
                });
            });
        }

        // --- DOM RENDERING ---
        function renderDashboard(selectedProvince) {
            const contentEl = document.getElementById('dashboard-content');
            contentEl.innerHTML = ''; // Clear previous content

            dimensions.forEach(dim => {
                const dimensionEl = document.createElement('section');
                dimensionEl.className = 'mb-8';
                
                const colorClasses = {
                    teal: 'border-t-teal-500',
                    blue: 'border-t-blue-500',
                    green: 'border-t-green-500',
                    purple: 'border-t-purple-500',
                    orange: 'border-t-orange-500'
                };

                let indicatorsHTML = '';
                dim.indicators.forEach(ind => {
                    const indicatorData = fullData.get(ind.id);
                    if (!indicatorData) return;

                    // Calculate rank
                    const sortedData = [...indicatorData].sort((a, b) => {
                        return ind.higherIsBetter ? b.value - a.value : a.value - b.value;
                    });
                    const rank = sortedData.findIndex(p => p.province === selectedProvince) + 1;
                    const provinceData = indicatorData.find(p => p.province === selectedProvince);
                    
                    if (!provinceData) return;

                    const scoreColor = provinceData.score >= 80 ? 'text-green-600' : provinceData.score >= 50 ? 'text-yellow-600' : 'text-red-600';

                    indicatorsHTML += `
                        <div class="kpi-card bg-white p-5 rounded-xl shadow-sm border-l-4 border-${dim.color}-500" 
                             data-indicator-id="${ind.id}" 
                             data-indicator-name="${ind.name}"
                             data-selected-province="${selectedProvince}">
                            <h4 class="font-semibold text-gray-600">${ind.name}</h4>
                            <div class="text-center my-4">
                                <span class="text-4xl font-bold text-gray-800">${provinceData.value.toLocaleString()}</span>
                                <span class="text-lg text-gray-500 ml-1">${ind.unit}</span>
                            </div>
                            <div class="flex justify-between items-center text-sm">
                                <div class="font-medium">
                                    <span>คะแนน:</span>
                                    <span class="text-lg font-bold ${scoreColor}">${provinceData.score}</span>
                                    <span class="text-gray-500">/ 100</span>
                                </div>
                                <div class="font-medium text-right">
                                    <span class="text-gray-500">ลำดับที่</span>
                                    <span class="text-lg font-bold text-${dim.color}-600">${rank}</span>
                                    <span class="text-gray-500">/ ${provinces.length}</span>
                                </div>
                            </div>
                        </div>
                    `;
                });

                dimensionEl.innerHTML = `
                    <div class="border-t-4 ${colorClasses[dim.color]} pt-2 mb-4">
                        <h2 class="text-2xl font-bold">${dim.title}</h2>
                    </div>
                    <div class="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 xl:grid-cols-4 gap-6">
                        ${indicatorsHTML}
                    </div>
                `;
                contentEl.appendChild(dimensionEl);
            });
            
            // Re-attach event listeners after rendering
            attachCardEventListeners();
        }

        function populateDropdown() {
            const selectEl = document.getElementById('province-select');
            provinces.forEach(province => {
                const option = document.createElement('option');
                option.value = province;
                option.textContent = province;
                selectEl.appendChild(option);
            });
            selectEl.value = 'กรุงเทพมหานคร'; // Set default
            selectEl.addEventListener('change', (e) => {
                renderDashboard(e.target.value);
            });
        }

        // --- TOOLTIP & CHART LOGIC ---
        const tooltipEl = document.getElementById('tooltip');
        const tooltipTitleEl = document.getElementById('tooltip-title');
        const chartCanvas = document.getElementById('tooltip-chart');
        let tooltipChart = null;

        function attachCardEventListeners() {
            const cards = document.querySelectorAll('.kpi-card');
            cards.forEach(card => {
                card.addEventListener('mouseenter', handleCardMouseEnter);
                card.addEventListener('mouseleave', handleCardMouseLeave);
            });
        }
        
        document.body.addEventListener('mousemove', (e) => {
            if (tooltipEl.style.display === 'block') {
                // Position tooltip relative to the viewport
                const tooltipRect = tooltipEl.getBoundingClientRect();
                let x = e.clientX + 15;
                let y = e.clientY + 15;

                // Adjust if tooltip goes off-screen
                if (x + tooltipRect.width > window.innerWidth) {
                    x = e.clientX - tooltipRect.width - 15;
                }
                if (y + tooltipRect.height > window.innerHeight) {
                    y = e.clientY - tooltipRect.height - 15;
                }
                
                tooltipEl.style.left = `${x}px`;
                tooltipEl.style.top = `${y}px`;
            }
        });

        function handleCardMouseEnter(event) {
            const card = event.currentTarget;
            const indicatorId = card.dataset.indicatorId;
            const indicatorName = card.dataset.indicatorName;
            const selectedProvince = card.dataset.selectedProvince;

            const indicatorData = fullData.get(indicatorId);
            if (!indicatorData) return;

            // Sort data for chart display (top 15 + selected province if not in top 15)
            const higherIsBetter = dimensions.flatMap(d => d.indicators).find(i => i.id === indicatorId).higherIsBetter;
            const sortedData = [...indicatorData].sort((a, b) => higherIsBetter ? b.value - a.value : a.value - b.value);
            
            const labels = sortedData.map(d => d.province);
            const data = sortedData.map(d => d.value);
            const backgroundColors = sortedData.map(d => d.province === selectedProvince ? '#EF4444' : '#3b82f6'); // Red for selected, blue for others
            const borderColors = sortedData.map(d => d.province === selectedProvince ? '#dc2626' : '#2563eb');

            tooltipTitleEl.textContent = indicatorName;
            tooltipEl.style.display = 'block';
            
            if (tooltipChart) {
                tooltipChart.destroy();
            }

            tooltipChart = new Chart(chartCanvas, {
                type: 'bar',
                data: {
                    labels: labels,
                    datasets: [{
                        label: indicatorName,
                        data: data,
                        backgroundColor: backgroundColors,
                        borderColor: borderColors,
                        borderWidth: 1
                    }]
                },
                options: {
                    indexAxis: 'y', // Horizontal bar chart
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: {
                            display: false
                        },
                        tooltip: {
                            callbacks: {
                                label: function(context) {
                                    return `${context.label}: ${context.raw.toLocaleString()}`;
                                }
                            }
                        }
                    },
                    scales: {
                        y: {
                            ticks: {
                                color: '#d1d5db' // Light gray for labels
                            },
                            grid: {
                                color: 'rgba(209, 213, 219, 0.2)'
                            }
                        },
                        x: {
                            ticks: {
                                color: '#d1d5db'
                            },
                            grid: {
                                color: 'rgba(209, 213, 219, 0.2)'
                            }
                        }
                    }
                }
            });
        }

        function handleCardMouseLeave() {
            tooltipEl.style.display = 'none';
            if (tooltipChart) {
                tooltipChart.destroy();
                tooltipChart = null;
            }
        }


        // --- INITIALIZATION ---

            generateMockData();

            renderDashboard('กรุงเทพมหานคร'); // Initial render
